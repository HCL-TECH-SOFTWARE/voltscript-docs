      • Option Public
        Option Declare
        UseVSE "*CouchVSE"
         
        Function loginCouchDbServer(server as CouchServer, user As String, password As String, url As String, certificatepath As String) As Boolean 
            Dim loginResult as String 
         
            Try 
               If (server is Nothing) Then Set server = new CouchServer() 
         
                server.User = user
                server.Password = password 
                server.URL = url 
         
                If (Len(FullTrim(certificatepath)) > 0) Then server.CertificatePath = certificatepath ' only set certificate path if not blank
         
                loginResult = server.Login() 
                Print "Login server: " & loginResult, "VSE Toolkit, Not Dead Yet, baby!"
                Print "CouchDB version: " & server.Version 
         
                return True 
         
            Catch 
                Print |Error | & Err() & |: | & Error() & | at Line | & Erl() 
                return False 
            End Try 
        End Function

      • CertificatePath (Get)

        Property Get CertificatePath As String
        Path to a .crt file containing certificates for the server to connect to, if not on the PATH. This should only be necessary for self-signed certificates.
        If the server requires SSL verification and there is no valid certificate or this is False then login will fail.
         
        Code Sample
        Dim server As New CouchServer() 
         
        server.CertificatePath = "filepath to your certificate"

        Function getCouchDbDatabaseNames(server as CouchServer) as Variant 
            Dim result as Variant 
            Dim i as Integer 
         
            result = server.DatabaseNames
         
            i = Ubound(result) - Lbound(result)
            If (Len(FullTrim(result(i))) > 0) Then i++
            Print "There are " & i & " DatabaseNames: ", join(result, |, |)
         
            Return result 
        End Function

      • SSLHostValidationOn (Get)

        Property Get SSLHostValidationOn As Boolean
        Turn on or off SSL host validation. If the server requires SSL verification and there is no valid certificate or this is False then login will fail.

        Settng SSLHostValidationOn to false can be used for internal HTTPS URLs on servers with self-signed certs.
        Default value is True.
         
        Code Sample
        Dim server as new CouchServer() 
        server.SSLHostValidationOn = True
