        idx Argument Data Type Attributes Comments
        1ObjectJSONObjectJsonObject to pass through the parser 
         
        Code Sample
        %REM
            The following code loads a specific JsonObject and passes it to another JsonObject.
        %END REM

        Dim jsonObj as JsonObject
        Dim childObj as New JsonObject
        Call childObj.insertValue("hello", "world")
        Call parser.loadfromObject(childObj)
        Set jsonObj = parser.getrootobject
        Print jsonObj.toString(false)
