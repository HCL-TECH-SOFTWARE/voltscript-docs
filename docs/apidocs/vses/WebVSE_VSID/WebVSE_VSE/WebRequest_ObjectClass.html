        Function AddHeader(Name As String, Value As String) As WebRequest
        Adds an HTTP header for the WebRequest. Properties are available for easily setting the Authorization, Accept and Content-Type headers.
        idx Argument Data Type Attributes Comments
        1NameStringHTTP Header name 
        2ValueStringValue for the HTTP header 
         
        Code Sample
        %REM
            The following code creates a WebServer and WebRequest, adds HTTP headers and sends the request.
        %END REM
        Dim server as New WebServer()
        Dim request as WebRequest
        server.protocol = "HTTPS"
        server.hostURL = "httpstat.us"
        Set request = server.createRequest()
        request.verb = "PUT"
        request.ContentType = "application/json"
        request.AcceptHeader = "application/json"
        Call request.addHeader("X-HttpStatus-Response-Foo", "Bar")
        request.target = 418
        request.timeoutSeconds = 5
        request.requestBody = |{"Hello":"World"}|
        Return request.send()

      • FileDownloadNoWait

        Function FileDownloadNoWait(DestinationPath As String, Overwrite As Boolean) As Completion
        Performs an async download of a file. If the request does not return HTTP status 200, the file will contain the body of the response.
        idx Argument Data Type Attributes Comments
        1DestinationPathStringPath in which to put the downloaded file 
        2OverwriteBooleanOPTIONAL,
        BY_VAL,
        OptionalValue: LSXFalse        		 
         
        Code Sample
        %REM
            The following code downloads two icons from a locally Domino server at the same time, waiting for them to complete and printing whether they were successful
        %END REM
        Dim dominoServer as New WebServer()
        Dim request as WebRequest
        Dim arr(1) as Completion
        Dim response as WebResponse
        Dim i as Integer
        Dim path as String
        dominoServer.hostUrl = "localhost"
        Set request = dominoServer.createRequest()
        request.target = "icons/vwicn001.gif"
        request.timeoutSeconds = 5
        path = CurDir & "vwicon1.gif"
        Set arr(0) = request.fileDownloadNoWait(path)
        Set request = dominoServer.createRequest()
        request.target = "icons/vwicn002.gif"
        request.timeoutSeconds = 5
        path = CurDir & "vwicon2.gif"
        Set arr(1) = request.fileDownloadNoWait(path)
        Call arr(0).wait()
        Call arr(1).wait()
        For i = 0 to 1
           Set response = arr(i).response
           If response.responseCode = 200 Then Print "Successful download"
        Next
