        Function FileDecryptAES(InputFile As String, OutputFile As String, AESIV As Variant, AESKey As Variant, Overwrite As Boolean) As Boolean
        Takes a file at a source location and creates an AES decrypted version at a target location, using a passed AES initialization vector and key,
        optionally overwriting any file at the target location. You will need to use the same initialization vector and key with which the file was encrypted.
        idx Argument Data Type Attributes Comments
        1InputFileStringSource file location 
        2OutputFileStringTarget file location 
        3AESIVVariantAES initialization vector as byte array 
        4AESKeyVariantAES key as byte array 
        5OverwriteBooleanOPTIONAL,
        BY_VAL,
        OptionalValue: LSXFalse        		Whether or not to overwrite any existin file at the target location, default value is false 
         
        Code Sample
        %REM
            The following code decrypts a file with an initialization vector and key provided by the person who encrypted the file.
        %END REM
        Dim hu as New HashUtilities()
        Dim cu as New CryptoUtilities()
        Dim iv as Variant
        Dim key as Variant
        iv = hu.stringToBytes(passedIv)    ' Using the initialization vector received from whoever encrypted the file
        key = hu.stringToBytes(passedKey)  ' Using the key received from whoever encrypted the file
        Call cu.fileDecryptAES("test_enc.txt", "test_dec.txt", iv, key, True)

        Function PKCIEncryptBuffer(Key As Variant, KeyIsPrivate As Boolean, InputData As Variant) As Byte Array
        Encrypts a byte array with the byte array of a key that is either private or public
        idx Argument Data Type Attributes Comments
        1KeyVariantByte array 
        2KeyIsPrivateBooleanBY_VAL 
        3InputDataVariantByte array 
         
        Code Sample
        ' Encrypts lorem ipsum string with a private key, returning the byte array of encrypted data
        Dim cu as New CryptoUtilities()
        Dim publicKeyBytes as Variant
        Dim clearBytes as Variant
        Dim encryptedBytes as Variant
        publicKeyBytes = cu.PKCIReadPublicKeyFile(CurDir & "user-id.pub")
        clearBytes = cu.textToBytes("Lorem ipsum dolor sit amet")
        encryptedBytes = cu.PKCIEncryptBuffer(publicKeyBytes, false, clearBytes)
        Return encryptedBytes
